$(function(){
  /*****准备数据***********/
    var container = $('.table-wrap');
    var blinkBtn = $('div.ns_container .reg');
    var gamemoney =  $('.gamemoney');
    var timer = null;
    var $js_acc = $("#reg_acc");//刷新查看余额（仅限会员客户）
    var loginRemainingTime = 0; //验证码短信接收时间
    var loginGetCodeBtn = $("#btn_loggetphcode");
    var _globalHiddens = $('#js-global-hiddens'); 
    var _globalHiddenValues = prj.getHiddenValues("#js-global-hiddens");
    loginRemainingTime = _globalHiddenValues.loginRemainingTime;
    $("ul#logSelect li").bind("click",function(){
         var _index = $(this).index();
         $(this).addClass("select");
         $(this).siblings().removeClass("select");
         $(".logSelectBox .sublist").eq(_index).removeClass("E03-none");
         $(".logSelectBox .sublist").eq(_index).siblings(".sublist").addClass("E03-none");
     })

  var pageContent = {
      init:function(){   
  		    var curNumber = tool.cookie('curNumber');//余额cookie	
          this._function.timeOfloginByPhone();
          this._function.topLoginByPhone();
          this._function.phoneHasFill();	
  		    setInterval(this._function.makeItBlink,500);// 注册按钮闪烁
    		  if(curNumber== ''){
              this._function.toWord();
          }else if(curNumber){
              $js_acc.text(tool.cookie('curNumber'));
          }else{      
              if(prj.getHiddenValues(_globalHiddens).customerType == '-1'){
                  tool.cookie('pcclick','');
              }else{
                  this._function.getCustomerMoney();
                  tool.cookie('curNumber');
                  $js_acc.text(curNumber);
              }
          }
          $("#btn_logout").bind("click",function(){
              tool.cookie('curNumber','');
          })    
          $('.js-regularloginName').on('click',this._function.getCustomerMoney);
      },

      _function:{
        // 获取游戏额度
        getCustomerMoney:function(){
        	  $js_acc.text('查询中');
          	$.ajax({
        				type:'POST',
        				url:"getGameCredit.htm",
        				data:{flag:2},
                dataType:'json',
        				success:function(data){
                  if(data){
                    if(data.status == 1){
                        $js_acc.text(data.credit);
                        tool.cookie('curNumber',data.credit);
                    }else if(data.status == 0){
                        new Window().init({
                            html:'当前用户已退出登录，请重新登录',
                            confirm:{
                              callback:function(){
                                  window.location.href = '/index.htm';
                              }
                            },
                            closed:function(){
                              window.location.href = '/index.htm';
                            }
                        });
                    }
                  }
      				},
      				error:function(data){
      					 prj.loadingcancel();
      				}
          	})        
        },
        phoneHasFill:function(){  
            $(document).on('change','#logphone',function(){
                var val = $(this).val();
                $.when(validModule.registphone({val:val,msgtype:'dialog',validtype:'login'})).done(function(data){
                    var regFlag = data.num;
                    if(loginRemainingTime<=0){
                        if( regFlag == 1||regFlag == 0){
                            loginGetCodeBtn.removeClass("disable");
                            tool.cookie('logmsgSend',1);
                            
                        }else{
                            loginGetCodeBtn.addClass("disable"); 
                            tool.cookie('logmsgSend',0);    
                        }                       
                    }else{
                        loginGetCodeBtn.addClass("disable"); 
                        tool.cookie('logmsgSend',0);
                    }
                })                
            });
        },
        makeItBlink:function(){
		      blinkBtn.toggleClass("makeBlink");
        },
        timeOfloginByPhone:function(){
            //全局变量 头部手机登录获取验证码
            if(loginRemainingTime){
                if(loginRemainingTime<=0){
                    // 可发送验证码
                    tool.cookie('logmsgSend',1);
                    loginGetCodeBtn.addClass("disable");
                }else{
                    //不能发送验证码
                    tool.cookie('logmsgSend',0);
                    tool.simpleDowntime({
                      container:loginGetCodeBtn,
                          timeDiff:loginRemainingTime,
                          prefix:'重新发送',
                          callback:function(){
                            loginGetCodeBtn.removeClass('disable').html('获取验证码');
                              tool.cookie('logmsgSend',1);
                          } 
                    });
                }
            }
        },
        // 登录操作
        topLoginByPhone:function(){
            //头部登陆获取验证码按钮点击
            // var loginGetCodeBtn = $("#btn_loggetphcode");

            loginGetCodeBtn.bind("click",function(){
                // var send_content="";
                var _this = $(this);
                var logphone = $("#logphone");
                $.when(validModule.registphone({val: logphone.val(), validtype: 'reg'})).done(function (data) {
                    if (data.num == 0) {
                        var send_content = 'smsreguser';
                    }else{
                        var send_content = 'smslogin';
                    }                 
                    // _this.addClass('disable');
                    if(!_this.hasClass('disable')){             
                        tool.singleWay('loginsendMsg-key',function(key){
                            var kit = this;
                            _this.addClass('disable');
                            tool.cookie('logmsgSend',0);
                            tool.ajax({
                                type:'post',
                                url:'SMSJson.htm',
                                data:{'operate':'sendsms','flag':'2','phone':logphone.val(),'send_content':send_content},
                                hasLoading:false,
                                success:function(res){
                                    // 发送成功
                                    if(res.status==1){
                                        timeDiff = res.timeRemaining;
                                         tool.simpleDowntime({
                                            container:$(_this),
                                            timeDiff:timeDiff,
                                            prefix:'重新发送',
                                            callback:function(){
                                                tool.cookie('logmsgSend',1);
                                                _this.removeClass('disable');
                                                _this.html('获取验证码');
                                                kit.resolve(key);
                                            }
                                        });
                                        $.dialog({
                                            html:res.mes
                                        })
                                    }else if(res.status==0){
                                        $.dialog({
                                          html:res.mes
                                        })
                                        _this.removeClass("disable");
                                        kit.resolve(key); 
                                    }
                                },
                                error:function(res){
                                    $.dialog({
                                      html:res.mes
                                    });
                                    kit.resolve(key);
                                }
                            });
                        });
                    }
               })
            })
        },
        toWord:function(e){
			     $js_acc.text("刷新查看");
        }
      }
  }
  pageContent.init(); 	
});
