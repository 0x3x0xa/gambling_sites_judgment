function pop_countDown(time,id){
  var day_elem1 = $(id);
  var end_time1 = new Date(time).getTime(),//月份是实际月份-1
  sys_second1 = (end_time1-new Date().getTime())/1000;
  var timer1 = setInterval(function(){
	if (sys_second1 > 1) {
	  sys_second1 -= 1;
	  var day1 = Math.floor((sys_second1 / 3600) / 24);
	  day_elem1 && $(day_elem1).text(day1+'天');//计算天
	}else{
	  clearInterval(timer1);
	}
  }, 1000);
}
