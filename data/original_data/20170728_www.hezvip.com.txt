//右側浮動選單(float.js)

var float_box = document.getElementById("float_r_ight"),
    float_box_open = $('#sidebar_open'),
    float_box_close = $('#sidebar_close'),
    float_box_main = $('#float_right_main'),
    float_box_log = $("#float_right_log"), fin, fout;

var flaot_box_in = function () {
    $(float_box).stop().animate({width:133,height:182},1000);
    float_box_open.hide();
    float_box_close.show();
    float_box_log.hide();
    float_box_main.show();
}
var flaot_box_out = function () {
    $(float_box).stop().animate({width:133,height:108},1000,function(){
        float_box_main.hide();
        float_box_log.show();
    });
    float_box_close.hide();
    float_box_open.show();
}

float_box_open.click(function () {
    flaot_box_in();
});

float_box_close.click(function () {
    flaot_box_out();
});

$(".sidebar_box_close").click(function () {
    $("#float-sidebar").remove();
