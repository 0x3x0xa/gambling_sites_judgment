                    function get_forum() {
                        $.post('/queryBbsTie.php', {
                            'type': '63'
                        }, function (data) {
                            if (data == '' || data == 'empty') {
                                return false;
                            } else {
                                var tmp1 = builHtml1(data);
                                var target1 = $('#j-forum-list');
                                target1.html(tmp1);
                            }

                        });

                        function builHtml1(data) {

                            var html = '',
                                tpl_forum = $('#tpl_forum').html(),
                                forum_url = '';
                            for (var i = 0; i < data.length; i++) {

                                var obj = data[i];
                                forum_url = 'http://bbs.qilebbs.com/thread-' + obj.id + '-1-1.html';

                                html += tpl_forum.replace(/\{\{createDate\}\}/g, obj.createDate)
                                    .replace(/\{\{num\}\}/g, i + 1)
                                    .replace(/\{\{forum_url\}\}/g, forum_url)
                                    .replace(/\{\{title\}\}/g, obj.title);
                            }

                            return html;
                        }
                    }
