  function specialAgentLogin() {
    var loginData = {
    	"user_name": $('#specialAgentName').val(),
    	"password": $('#specialAgentPwd').val(),
    	"verify_code": $('#specialAgentVerifyCode').val(),
    	"is_special_agent": 1

      type: "POST",
      url: "/Base/login.html",
      data: loginData,
      cache:false,
      dataType:"json",
      async:false,
      beforeSend: function () {
          // 禁用按钮防止重复提交，发送前响应

          $(".header_error").fadeIn();
          $(".header_error").fadeOut(1000);
          replaceSpecialAgentCode();
        }
      },
      complete: function () {
        //完成响应
        $('#specialAgentLoginBtn').removeAttr("disabled");
      }
    });
  }

        function getRandomArrayElements(arr, count) {
            var shuffled = arr.slice(0), i = arr.length, min = i - count, temp, index;
            while (i-- > min) {
                index = Math.floor((i + 1) * Math.random());
                temp = shuffled[index];
                shuffled[index] = shuffled[i];
                shuffled[i] = temp;
            }
            return shuffled.slice(min);
        }
