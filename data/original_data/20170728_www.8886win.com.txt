        function mobile_device_detect(url) {
            var patm = GetQueryString("patm");
            var isToH5 = "true";
            if ((patm == null || patm.toString().length <= 0) && isToH5 == "true") {
                if (url.indexOf("?") > 0) {
                    url = url + "&djSite=" + window.location.host + "&sourceId=0";
                } else {
                    url = url + "?djSite=" + window.location.host + "&sourceId=0";
                }
                var thisOS = navigator.platform;
                var os = new Array("iPhone", "iPod", "iPad", "android", "Nokia", "SymbianOS", "Symbian", "Windows Phone", "Phone", "Linux armv71", "MAUI", "UNTRUSTED/1.0", "Windows CE", "BlackBerry", "IEMobile");
                for (var i = 0; i < os.length; i++) {
                    if (thisOS.match(os[i])) {
                        window.location = url;
                    }
                }
                //因为相当部分的手机系统不知道信息,这里是做临时性特殊辨认
                if (navigator.platform.indexOf('iPad') != -1) {
                    window.location = url;
                }
                //做这一部分是因为Android手机的内核也是Linux
                //但是navigator.platform显示信息不尽相同情况繁多,因此从浏览器下手，即用navigator.appVersion信息做判断
                var check = navigator.appVersion;
                if (check.match(/linux/i)) {
                    //X11是UC浏览器的平台 ，如果有其他特殊浏览器也可以附加上条件
                    if (check.match(/mobile/i) || check.match(/X11/i)) {
                        window.location = url;
                    }
                }
                //类in_array函数
                Array.prototype.in_array = function (e) {
                    for (i = 0; i < this.length; i++) {
                        if (this[i] == e)
                            return true;
                    }
                    return false;
                }
            }
        }

        function GetQueryString(name) {
            var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)");
            var r = window.location.search.substr(1).match(reg);
            if (r != null) return unescape(r[2]); return null;
        }

        function convertCoreApiData(source) {
            if(!source)
                return [];

            var sourceData=JSON.parse(source);
            if(!sourceData.Datas||sourceData.length==0)
                return [];
            
            var result=[];
            for (var i in sourceData.Datas) {
                var item=sourceData.Datas[i];
                							                
                result.push({id:item[0],title:item[1],isEnable:item[2],
                    contentType:item[3],content:item[4],createTime:item[5],
                    startTime:item[6],endTime:item[7]})
            }

            return result;
        }

        function bindReminder(data) {
            $("#reminder-title").html(data.title);
            var lines=[];
            var value=data.content;
            var isDecimal=true;

            //如果找不到#分隔符，则证明是单独一行
            if (value.indexOf("#") == -1) {
                lines.push(value);
                isDecimal=false;
            }
            else {
                var values = [];

                if (value.startsWith("##")) {
                    values = value.split("#").filter(function (arg) {
                        return arg.length > 0
                    });
                }
                    //如果不是以##开头，则用默认它是用#，所以用#分割成字符数组，并显示
                else {
                    isDecimal=false;
                    values = value.split("#").filter(function (arg) {
                        return arg.length > 0
                    });
                }

                values.forEach(function (it) {
                    lines.push(it);
                });
            }

            if (lines.length==0) {
                return;
            }

            var content=$("#reminder-content");
            for (var i = 0; i < lines.length; i++) {
                var index=i+1;
                
                var item = $('<p  style="text-indent: 15px; padding-top: '+(index*5)+'px;"></p>');                
                if(isDecimal)
                    item.html(index+"."+lines[i]);
                else
                    item.html(lines[i]);

                content.append(item);
            }
        }
